	\begin{enumerate}

}

	\item #1
}

	Conditions:

}

	\item #1
}

	\\
	& Tested parameter: #1
}

	\\
	& Result: success
	\stepcounter{overallTest}
	\stepcounter{successTest}
}
\newcommand{\testResultFalse}[0] {
	Result: failed
	\stepcounter{overallTest}
}

	\end{itemize}
}

	\end{enumerate}
	\end{tabular*}\\

	\setcounter{overallTest}{0}
	\setcounter{successTest}{0}
}
